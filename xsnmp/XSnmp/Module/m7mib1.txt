META7-MIB DEFINITIONS ::= BEGIN

	IMPORTS
		enterprises		FROM RFC1155-SMI
		DisplayString	FROM RFC1213;



   Meta7 OBJECT IDENTIFIER ::= { enterprises 1212 }

   Mmr			OBJECT IDENTIFIER ::= {Meta7 1 }
   appGoup			OBJECT IDENTIFIER ::= {Mmr 2 }
   Meta7-Boot-Events		OBJECT IDENTIFIER ::= {Mmr 3}
   Meta7-App-Events		OBJECT IDENTIFIER ::= {Mmr 4}
--------------
--
--
--------------

   simName OBJECT-TYPE
	SYNTAX DisplayString
	ACCESS not-accessible
	STATUS mandatory
	DESCRIPTION "The name of the region."
	::= { appGroup 1 }



   appError OBJECT-TYPE
	SYNTAX DisplayString
	ACCESS not-accessible
	STATUS mandatory
	DESCRIPTION "The last known error."
	::= { appGroup 2 }

   bootSequence OBJECT-TYPE
	SYNTAX INTEGER {
		bootInitialised(1),
		regionLoaded(2),
		regionRegistration(3),
		mapGeneration(4),
		WatchDogUp(5),
		regionUp(6)
	}
	ACCESS not-accessible
	STATUS mandatory
	DESCRIPTION
		"Contains the actual step of the boot process."
	::= { appGroup 3 }

-------------------------
-- Traps
--
-------------------------
bootStatus TRAP-TYPE
	ENTERPRISE Meta7-Boot-Events
	VARIABLES { simName, bootSequence }
	--&ACTIONS { region name, bootsteps }
	--&MESG   "$1 Error: $2"
	DESCRIPTION "The sim $1 is booting. Actual step $2 "
	::= 2

bootCold TRAP-TYPE
	ENTERPRISE Meta7-Boot-Events
	VARIABLES { simName, appError }
	DESCRIPTION "ColdBoot Event"
	::= 1

TrapError TRAP-TYPE
	ENTERPRISE Meta7-App-Events
	VARIABLES { simName, appError }
	DESCRIPTION "The sim  exit on error "
	::= 1

watchDog TRAP-TYPE
	ENTERPRISE Meta7-App-Events
	VARIABLES { simName, appError }
	DESCRIPTION "Watchdog error trigged "
	::= 2

SimDown TRAP-TYPE
	ENTERPRISE Meta7-App-Events
	VARIABLES { simName }
	DESCRIPTION "SimLink Down"
	::= 3

SimUp TRAP-TYPE
	ENTERPRISE Meta7-App-Events
	VARIABLES { simName }
	DESCRIPTION "SimLink Up"
	::= 4
END
